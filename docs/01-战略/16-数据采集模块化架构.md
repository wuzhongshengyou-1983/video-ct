## 一、核心设计理念

```
                    ┌──────────────────────────────────┐
                    │         数据总线 (DataBus)         │
                    │   统一的 VideoCTContext 数据契约    │
                    └──────────┬───────────────────────┘
                               │
          ┌────────────────────┼────────────────────┐
          │                    │                    │
   ┌──────▼──────┐    ┌───────▼───────┐    ┌───────▼──────┐
   │  STEP 1     │    │   STEP 2      │    │   STEP 3     │
   │ 数据采集层   │ ──▶│  整理分析层    │ ──▶│  诊断输出层   │
   │             │    │               │    │              │
   │ 6 类模块    │    │ 4 个分析引擎   │    │ 7 份输出报告  │
   └─────────────┘    └───────────────┘    └──────────────┘
```

### 设计原则

| 原则 | 说明 |
|------|------|
| **模块热插拔** | 每个数据源实现统一接口，更换数据源不改业务代码 |
| **优先级链** | 每个模块族内多个实现按优先级排序，首选不可用时自动降级 |
| **健康感知** | 每个模块自报健康状态，Registry 缓存 30 秒，避免频繁探测 |
| **异常隔离** | 任一模块失败不影响其他模块，Pipeline 记录降级信息后继续 |
| **全开源** | 所有方案均为 GitHub 可验证的真开源项目，或「开源模型+托管API」模式 |
| **配置驱动** | 模块优先级通过配置文件控制，运维人员可直接调整 |

```python
from abc import ABC, abstractmethod

class PlatformDataSource(ABC):
    """平台元数据源——获取视频的播放数据、作者信息、评论等。"""

    @property
    @abstractmethod
    def source_id(self) -> str:
        """模块唯一标识，如 'mediacrawler' / 'shilonglee' / 'video_parser'."""
        ...

    @property
    @abstractmethod
    def priority(self) -> int:
        """优先级。数字越小越优先。1=首选, 2=降级, 3=兜底."""
        ...

    @property
    @abstractmethod
    def supported_platforms(self) -> list[str]:
        """该数据源覆盖的平台列表."""
        ...

    @abstractmethod
    async def fetch_video_meta(self, video_url: str) -> PlatformMeta:
        """根据视频链接获取完整元数据."""
        ...

    @abstractmethod
    async def fetch_comments(self, video_url: str, limit: int = 100) -> list[dict]:
        """获取视频评论."""
        ...

    @abstractmethod
    async def fetch_creator_videos(self, creator_id: str, limit: int = 20) -> list[PlatformMeta]:
        """获取创作者最近视频列表."""
        ...

    @abstractmethod
    async def health_check(self) -> HealthStatus:
        """自检当前数据源是否可用."""
        ...
```

```python
# fire_eye/adapters/platform/mediacrawler_adapter.py

import httpx
from fire_eye.contracts import PlatformMeta, CreatorInfo, HealthStatus

class MediaCrawlerAdapter(PlatformDataSource):
    source_id = "mediacrawler"
    priority = 1
    supported_platforms = ["douyin", "bilibili", "kuaishou", "xiaohongshu", "weibo"]

    def __init__(self, base_url: str = "http://mediacrawler:8080"):
        self._client = httpx.AsyncClient(base_url=base_url, timeout=30)

    async def fetch_video_meta(self, video_url: str) -> PlatformMeta:
        platform = self._detect_platform(video_url)
        resp = await self._client.post(f"/api/{platform}/video", json={
            "url": video_url, "extract_metadata": True, "extract_comments": False,
        })
        data = resp.json()
        return PlatformMeta(
            platform=platform,
            video_id=data["video_id"],
            title=data["title"],
            description=data.get("desc", ""),
            duration_sec=data.get("duration", 0),
            cover_url=data["cover_url"],
            play_count=data["stats"]["play_count"],
            like_count=data["stats"]["like_count"],
            comment_count=data["stats"]["comment_count"],
            share_count=data["stats"]["share_count"],
            collect_count=data["stats"]["collect_count"],
            creator=CreatorInfo(
                nickname=data["author"]["nickname"],
                follower_count=data["author"]["follower_count"],
            ),
            tags=data.get("tags", []),
            hashtags=data.get("hashtags", []),
            _source_id="mediacrawler",
        )

    async def fetch_comments(self, video_url: str, limit: int = 100) -> list[dict]:
        platform = self._detect_platform(video_url)
        resp = await self._client.post(f"/api/{platform}/comments", json={
            "url": video_url, "limit": limit,
        })
        return resp.json()["comments"]

    async def fetch_creator_videos(self, creator_id: str, limit: int = 20) -> list[PlatformMeta]:
        resp = await self._client.post(f"/api/douyin/creator", json={
            "creator_id": creator_id, "limit": limit,
        })
        return [self._to_meta(v) for v in resp.json()["videos"]]

    async def health_check(self) -> HealthStatus:
        try:
            r = await self._client.get("/health")
            return HealthStatus.HEALTHY if r.status_code == 200 else HealthStatus.DOWN
        except Exception:
            return HealthStatus.DOWN

    def _detect_platform(self, url: str) -> str:
        import re
        patterns = {
            "douyin": r"douyin\.com|iesdouyin\.com",
            "bilibili": r"bilibili\.com|b23\.tv",
            "kuaishou": r"kuaishou\.com",
            "xiaohongshu": r"xiaohongshu\.com|xhslink\.com",
            "weibo": r"weibo\.com|t\.cn",
        }
        for plat, pat in patterns.items():
            if re.search(pat, url): return plat
        return "unknown"

    def _to_meta(self, v: dict) -> PlatformMeta: ...
```

### 实现 1：FFprobe（FFmpeg ⭐47,000+，LGPL）

当前 `video_tech.py` 已经集成了 `get_video_tech_basics()`，v2.5 在此基础上增强字段。

```python
# fire_eye/adapters/tech/ffprobe_source.py

import json, subprocess
from fire_eye.contracts import TechMeta, HealthStatus

class FFprobeSource(TechMetadataSource):
    source_id = "ffprobe"
    priority = 1

    async def extract(self, video_url: str) -> TechMeta:
        # 复用现有 video_tech.py 方法
        from app.services.video_tech import resolve_real_stream_url

        real_url = resolve_real_stream_url(video_url) or video_url
        result = subprocess.run(
            ["ffprobe", "-v", "quiet", "-print_format", "json",
             "-show_format", "-show_streams", real_url],
            capture_output=True, text=True, timeout=30,
        )
        info = json.loads(result.stdout)

        video_stream = next(
            (s for s in info["streams"] if s["codec_type"] == "video"), {}
        )
        audio_streams = [s for s in info["streams"] if s["codec_type"] == "audio"]
        fmt = info.get("format", {})

        meta = TechMeta(
            resolution=f"{video_stream.get('width', 0)}x{video_stream.get('height', 0)}",
            codec=video_stream.get("codec_name", "unknown"),
            profile=video_stream.get("profile", "unknown"),
            bitrate_kbps=int(fmt.get("bit_rate", 0)) // 1000 if fmt.get("bit_rate") else None,
            fps=round(eval(str(video_stream.get("r_frame_rate", "0/1"))), 1),
            duration_seconds=round(float(fmt.get("duration", 0)), 1),
            pix_fmt=video_stream.get("pix_fmt", "unknown"),
            color_space=video_stream.get("color_space", "unknown"),
            gop_size=self._detect_gop(real_url),
            audio_codec=audio_streams[0].get("codec_name") if audio_streams else None,
            audio_bitrate_kbps=int(audio_streams[0].get("bit_rate", 0)) // 1000 if audio_streams and audio_streams[0].get("bit_rate") else None,
            file_size_mb=round(int(fmt.get("size", 0)) / (1024 * 1024), 1) if fmt.get("size") else None,
            has_b_frames=video_stream.get("has_b_frames", 0) > 0,
            _source_id="ffprobe",
        )
        # 计算技术质量评分
        meta.tech_score, meta.tech_issues = self._compute_tech_score(meta)
        return meta

    def _detect_gop(self, url: str) -> int | None:
        """检测关键帧间隔（GOP大小）."""
        try:
            result = subprocess.run(
                ["ffprobe", "-v", "quiet", "-show_frames", "-read_intervals", "0%+#120",
                 "-select_streams", "v:0", "-show_entries", "frame=key_frame", url],
                capture_output=True, text=True, timeout=30,
            )
            return result.stdout.count("key_frame=1")
        except Exception:
            return None

    def _compute_tech_score(self, meta: TechMeta) -> tuple[int, list[str]]:
        """基于赛道标准的视频技术质量评分."""
        score, issues = 100, []
        if meta.bitrate_kbps and meta.bitrate_kbps < 3000:
            score -= 15
            issues.append(f"码率{meta.bitrate_kbps}kbps低于建议值3000kbps")
        if meta.resolution and "1080" not in meta.resolution:
            score -= 10
            issues.append(f"分辨率{meta.resolution}未达1080P")
        if meta.fps and meta.fps < 24:
            score -= 10
            issues.append(f"帧率{meta.fps}fps低于24fps")
        return max(0, score), issues

    async def health_check(self) -> HealthStatus:
        try:
            subprocess.run(["ffprobe", "-version"], capture_output=True, timeout=5)
            return HealthStatus.HEALTHY
        except Exception:
```

```python
# fire_eye/adapters/content/qwen_vl_pipeline.py

import asyncio, json, subprocess, tempfile, os
from openai import OpenAI
from fire_eye.contracts import ContentAnalysis, HealthStatus

class QwenVLContentPipeline(VideoContentAISource):
    """基于 Qwen3-VL + faster-whisper 的开源视频内容理解 Pipeline."""

    source_id = "qwen_vl_pipeline"
    priority = 1

    def __init__(self, mode: str = "api", api_key: str = "", base_url: str = ""):
        self._mode = mode
        if mode == "api":
            self._vision_client = OpenAI(
                base_url=base_url or "https://api.siliconflow.cn/v1",
                api_key=api_key,
            )

    async def analyze(self, video_url: str) -> ContentAnalysis:
        # Step 1: 抽帧（每3秒1帧，最多20帧）
        frames = await self._extract_keyframes(video_url, interval=3, max_frames=20)

        # Step 2: 并行视觉理解
        frame_analyses = await asyncio.gather(*[
            self._analyze_frame(f) for f in frames
        ])

        # Step 3: 语音转写
        transcript = await self._transcribe_audio(video_url)

        # Step 4: 融合为 ContentAnalysis
        return self._merge(frame_analyses, transcript)

    async def analyze_quick(self, video_url: str) -> ContentAnalysis:
        frames = await self._extract_keyframes(video_url, interval=5, max_frames=6)
        frame_analyses = await asyncio.gather(*[
            self._analyze_frame(f) for f in frames
        ])
        transcript = await self._transcribe_audio(video_url)
        return self._merge(frame_analyses, transcript)

    async def _extract_keyframes(self, video_url: str, interval: int, max_frames: int) -> list[str]:
        with tempfile.TemporaryDirectory() as tmpdir:
            pattern = os.path.join(tmpdir, "frame_%03d.jpg")
            subprocess.run(
                ["ffmpeg", "-y", "-timeout", "15000000", "-i", video_url,
                 "-vf", f"fps=1/{interval},scale=640:-1",
                 "-frames:v", str(max_frames), "-q:v", "3", pattern],
                capture_output=True, text=True, timeout=60,
            )
            return sorted([os.path.join(tmpdir, f) for f in os.listdir(tmpdir)
                          if f.startswith("frame_") and f.endswith(".jpg")])

    async def _analyze_frame(self, frame_path: str) -> dict:
        import base64
        with open(frame_path, "rb") as f:
            img_b64 = base64.b64encode(f.read()).decode()

        if self._mode == "api":
            resp = self._vision_client.chat.completions.create(
                model="Qwen/Qwen2.5-VL-72B-Instruct",
                messages=[{"role": "user", "content": [
                    {"type": "image_url", "image_url": {"url": f"data:image/jpeg;base64,{img_b64}"}},
                    {"type": "text", "text": "描述这个画面：场景类型、人物数量与动作、文字内容、光线质量、构图方式。返回JSON。"},
                ]}],
                response_format={"type": "json_object"},
            )
            return json.loads(resp.choices[0].message.content)
        else:
            return await self._local_vl_inference(frame_path)

    async def _transcribe_audio(self, video_url: str) -> list[dict]:
        """faster-whisper 语音转写."""
        audio_path = await self._extract_audio(video_url)
        # 使用 faster-whisper Python SDK
        from faster_whisper import WhisperModel
        model = WhisperModel("large-v3", device="cuda", compute_type="float16")
        segments, _ = model.transcribe(audio_path, language="zh")
        return [{"start": s.start, "end": s.end, "text": s.text} for s in segments]

    def _merge(self, frame_analyses: list[dict], transcript: list[dict]) -> ContentAnalysis:
        shots = len(frame_analyses)
        return ContentAnalysis(
            timeline=self._build_timeline(frame_analyses, transcript),
            summary=self._summarize(frame_analyses, transcript),
            transcript=transcript,
            speech_rate_wps=self._calc_speech_rate(transcript),
            shot_count=shots,
            avg_shot_duration_sec=self._calc_avg_shot(frame_analyses),
            events=self._detect_events(frame_analyses, transcript),
            hook_point_sec=self._find_hook(transcript, frame_analyses),
            cta_point_sec=self._find_cta(transcript),
            visual_quality_score=self._assess_quality(frame_analyses),
            _source_id="qwen_vl_pipeline",
        )

    # ── 分析辅助方法 ──
    def _build_timeline(self, frames, transcript) -> list[dict]: ...
    def _summarize(self, frames, transcript) -> str: ...
    def _calc_speech_rate(self, transcript) -> float: ...
    def _calc_avg_shot(self, frames) -> float: ...
    def _detect_events(self, frames, transcript) -> list[dict]: ...
    def _find_hook(self, transcript, frames) -> float | None: ...
    def _find_cta(self, transcript) -> float | None: ...
    def _assess_quality(self, frames) -> float: ...
    async def _extract_audio(self, video_url: str) -> str: ...
    async def _local_vl_inference(self, frame_path: str) -> dict: ...

    async def health_check(self) -> HealthStatus:
        try:
            if self._mode == "api":
                resp = self._vision_client.models.list()
                return HealthStatus.HEALTHY if resp else HealthStatus.DOWN
            return HealthStatus.HEALTHY
        except Exception:
            return HealthStatus.DOWN
```

```python
import clickhouse_connect

class ShakaClickHouseQoESource(QoEDataSource):
    source_id = "shaka_clickhouse"
    priority = 1

    def __init__(self, ch_host: str, ch_port: int):
        self._ch = clickhouse_connect.get_client(host=ch_host, port=ch_port)

    async def query_playback_metrics(self, video_id: str,
                                     time_range_hours: int = 24) -> QoEProfile:
        result = self._ch.query(f"""
            SELECT
                quantile(0.5)(ttff_ms) AS p50_ttff,
                countIf(event='buffer') / greatest(countIf(event='play'), 1) AS rebuffer_rate,
                countIf(event='error') / greatest(count(DISTINCT session_id), 1) AS error_rate,
                quantile(0.5)(bitrate_kbps) AS p50_bitrate,
                quantile(0.5)(completion_rate) AS p50_completion
            FROM playback_events
            WHERE video_id = '{video_id}'
              AND timestamp >= now() - INTERVAL {time_range_hours} HOUR
        """)
        row = result.first_row
        return QoEProfile(
            ttff_ms=row[0],
            rebuffer_rate_pct=round(row[1] * 100, 2) if row[1] else None,
            error_rate_pct=round(row[2] * 100, 2) if row[2] else None,
            avg_bitrate_kbps=row[3],
            completion_rate_pct=round(row[4] * 100, 2) if row[4] else None,
            _source_id="shaka_clickhouse",
        )

    async def health_check(self) -> HealthStatus:
        try:
            self._ch.query("SELECT 1")
            return HealthStatus.HEALTHY
```

```python
    """数据源注册中心——管理所有模块族的注册、调度、降级。

    def register_llm_backend(self, backend) -> None:
        self._llm_backends.append(backend)
        self._llm_backends.sort(key=lambda b: b.priority)

    def register_qoe_source(self, source) -> None:
        self._qoe_sources.append(source)
        self._qoe_sources.sort(key=lambda s: s.priority)

    # ── 调度接口：自动优先级 + 健康检查 + 降级 ──

    async def fetch_platform_meta(self, video_url: str):
        """获取平台元数据——自动选择最佳可用数据源."""
        from fire_eye.contracts import detect_platform
        platform = detect_platform(video_url)
        sources = self._platform_sources.get(platform, [])

        for source in sources:
            if not await self._is_healthy(source):
                continue
            try:
                meta = await source.fetch_video_meta(video_url)
                meta._source_id = source.source_id
                meta._source_priority = source.priority
                return meta
            except Exception as e:
                logger.warning(f"[Registry] {source.source_id} 调用失败: {e}")
                self._record_failure(source.source_id)
                continue

        from fire_eye.contracts import AllSourcesExhaustedError
        raise AllSourcesExhaustedError(f"平台 {platform} 所有数据源均不可用")

    async def extract_tech_meta(self, video_url: str):
        for source in self._tech_sources:
            if not await self._is_healthy(source):
                continue
            try:
                return await source.extract(video_url)
            except Exception as e:
                logger.warning(f"[Registry] {source.source_id} 提取失败: {e}")
                continue
        return None  # 技术元数据非必需

    async def analyze_content(self, video_url: str):
        for source in self._ai_sources:
            if not await self._is_healthy(source):
                continue
            try:
                result = await source.analyze(video_url)
                return result
            except Exception as e:
                logger.warning(f"[Registry] {source.source_id} AI分析失败: {e}")
                self._record_failure(source.source_id)
                continue
        return None  # AI分析非必需，可降级为纯文本诊断

    async def fetch_benchmarks(self, track: str, platform: str):
        for source in self._trend_sources:
            if not await self._is_healthy(source):
                continue
            try:
                videos = await source.fetch_trending(platform, track, limit=100)
                return self._compute_benchmarks(videos, track)
            except Exception as e:
                logger.warning(f"[Registry] {source.source_id} 趋势采集失败: {e}")
                continue
        return None

    async def diagnose_with_llm(self, ctx) -> dict:
        for backend in self._llm_backends:
            if not await self._is_healthy(backend):
                continue
            try:
                return await backend.diagnose(ctx)
            except Exception as e:
                logger.warning(f"[Registry] {backend.backend_id} 诊断失败: {e}")
                self._record_failure(backend.backend_id)
                continue
        raise AllSourcesExhaustedError("所有 LLM 后端均不可用")

    async def fetch_qoe(self, platform: str, video_id: str):
        for source in self._qoe_sources:
            if not await self._is_healthy(source):
                continue
            try:
                return await source.query_playback_metrics(video_id)
            except Exception as e:
                logger.warning(f"[Registry] {source.source_id} QoE查询失败: {e}")
                continue
        return None

    # ── 健康检查 ──

    async def _is_healthy(self, source) -> bool:
        sid = getattr(source, 'source_id', None) or getattr(source, 'backend_id', None)
        now = time.time()
        if sid in self._health_cache:
            if now - self._health_cache_ts.get(sid, 0) < self._health_interval:
                return self._health_cache[sid] == HealthStatus.HEALTHY

        try:
            status = await source.health_check()
            self._health_cache[sid] = status
            self._health_cache_ts[sid] = now
            if status == HealthStatus.DOWN:
                logger.warning(f"[Registry] {sid} 健康检查: DOWN")
            return status == HealthStatus.HEALTHY
        except Exception:
            self._health_cache[sid] = HealthStatus.DOWN
            self._health_cache_ts[sid] = now
            return False

    def _record_failure(self, source_id: str) -> None:
        self._failure_counts[source_id] = self._failure_counts.get(source_id, 0) + 1
        if self._failure_counts[source_id] >= 3:
            self._health_cache[source_id] = HealthStatus.DEGRADED
            logger.warning(f"[Registry] {source_id} 连续失败{self._failure_counts[source_id]}次，标记为 DEGRADED")

    def _compute_benchmarks(self, videos: list, track: str):
        """从竞品视频列表动态计算赛道基准."""
        from fire_eye.contracts import BenchmarkContext, PercentileBundle
        import numpy as np
        if len(videos) < 30:
            return None
        plays = [v.play_count for v in videos]
        likes = [v.like_count for v in videos]
        comments = [v.comment_count for v in videos]
        shares = [v.share_count for v in videos]
        collects = [v.collect_count for v in videos]

        def _pcts(arr):
            return PercentileBundle(
                p25=np.percentile(arr, 25), p50=np.percentile(arr, 50),
                p75=np.percentile(arr, 75), p90=np.percentile(arr, 90),
            )

        return BenchmarkContext(
            track=track,
            sample_size=len(videos),
            updated_at=datetime.now(),
            metrics={
                "点赞率": _pcts([(l / max(p, 1)) * 100 for l, p in zip(likes, plays)]),
                "评论率": _pcts([(c / max(p, 1)) * 100 for c, p in zip(comments, plays)]),
                "转发率": _pcts([(s / max(p, 1)) * 100 for s, p in zip(shares, plays)]),
                "收藏率": _pcts([(c / max(p, 1)) * 100 for c, p in zip(collects, plays)]),
            },
        )
```

```python
def _detect(url: str) -> str:
    import re
    patterns = {

        "bilibili": r"bilibili\.com|b23\.tv",
        "kuaishou": r"kuaishou\.com",
        "xiaohongshu": r"xiaohongshu\.com|xhslink\.com",
        "weibo": r"weibo\.com|t\.cn",

    }
    for plat, pat in patterns.items():
        if re.search(pat, url): return plat
```

## 十六、开源资源评审标准（多维调研框架）

### 16.1 七维评审标准

每个维度 0–10 分，满分 70。

| 维度 | 权重 | 评分标准 |
|------|------|---------|
| **① 合规性** | ×2.0 | 10=官方授权API；7=合规第三方；4=灰色地带爬虫；0=明确违规 |
| **② 数据质量** | ×1.5 | 10=字段完整+误差<5%；7=关键字段齐全；4=核心字段缺失或延迟>2h；0=数据不可信 |
| **③ 接入成本** | ×1.0 | 10=开箱即用HTTP调用；7=需注册企业认证；4=需自建服务；0=需自部署GPU/Docker |
| **④ 运维复杂度** | ×1.0 | 10=无状态纯API；7=需维护Token刷新；4=需维护登录态/Cookie；0=需维护本地服务 |
| **⑤ 覆盖广度** | ×1.0 | 10=覆盖全部目标平台；7=覆盖主要平台(≥3)；4=仅1-2平台；0=不覆盖目标 |
| **⑥ 实时性** | ×1.0 | 10=实时(<5min)；7=准实时(<1h)；4=延迟(2-24h)；0=批量离线(>24h) |
| **⑦ 单次成本** | ×0.5 | 10=免费/开源；8=<¥0.01/次；5=¥0.01-0.1/次；2=¥0.1-1/次；0=>¥1/次 |

**加权总分计算**：`总分 = Σ(维度分 × 权重)` ，满分 70，**≥50 推荐，40-49 备选，<40 淘汰**。

### 16.4 选型决策流程（每次引入新模块时执行）

```
Step 1  定位族群 → 确认属于 Family 1-6 中的哪一族
Step 2  七维打分 → 逐项评分，计算加权总分
Step 3  门槛判断 → 总分≥50 推荐 / 40-49 备选 / <40 直接淘汰
Step 4  合规核查 → ① 合规性 得分 < 7 则一票否决，不看总分
Step 5  成本测算 → 代入 16.3 成本模型，确认单次成本可接受
Step 6  实测验证 → 用3条真实视频跑通，记录实际耗时和数据质量
Step 7  写入文档 → 更新第十五节兼容性总表，注明测试日期和版本
```
