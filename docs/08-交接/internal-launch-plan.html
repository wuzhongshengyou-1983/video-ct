# services/api/app/services/auth_service.py
# 在现有文件中追加以下方法：

async def register_by_email(email: str, password: str, db: Session) -> User:
    """邮箱注册（内部测试用）"""
    if db.query(User).filter(User.email == email).first():
        raise BizException("EMAIL_EXISTS", "邮箱已注册")
    hashed = bcrypt.hashpw(password.encode(), bcrypt.gensalt()).decode()
    user = User(email=email, password_hash=hashed, nickname=email.split("@")[0])
    db.add(user)
    db.commit()
    return user

async def login_by_email(email: str, password: str, db: Session) -> str:
    """邮箱登录，返回 access_token"""
    user = db.query(User).filter(User.email == email).first()
    if not user or not bcrypt.checkpw(password.encode(), user.password_hash.encode()):
        raise BizException("INVALID_CREDENTIALS", "邮箱或密码错误")
    return create_access_token({"sub": str(user.id), "email": user.email})
# services/api/app/routers/auth.py
# 追加两个路由：

@router.post("/register/email")
async def register_email(data: EmailRegisterRequest, db: Session = Depends(get_db)):
    user = await auth_service.register_by_email(data.email, data.password, db)
    token = await auth_service.login_by_email(data.email, data.password, db)
    return {"access_token": token, "user_id": user.id}

@router.post("/login/email")
async def login_email(data: EmailLoginRequest, db: Session = Depends(get_db)):
    token = await auth_service.login_by_email(data.email, data.password, db)
    return {"access_token": token}
